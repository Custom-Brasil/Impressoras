140static const wchar_t* CcwResultToText(CcwResult result)
141{
142 DWORD i;
143
144 for (i = 0; i<ERROR_CCW_LISTSIZE; i++)
145 {
146 if (ERROR_CCW_LIST[i].ccwError == result)
147 return ERROR_CCW_LIST[i].strDescription;
148 }
149
150 return L"Unknown Error";
151}
